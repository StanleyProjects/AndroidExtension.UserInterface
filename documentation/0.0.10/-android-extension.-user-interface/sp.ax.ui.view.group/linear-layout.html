Builds an instance of LinearLayout by context and parameters or default. Sample:

fun foo(context: Context): LinearLayout {
    return linearLayout(
        context = context,
        layoutParams = ViewGroup::class.layoutParams(size = 5),
        background = ColorDrawable(Color.GREEN)
    )
}
