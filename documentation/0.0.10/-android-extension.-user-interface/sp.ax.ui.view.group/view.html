Builds an instance of View by parameters or default and adds to LinearLayout receiver if needToAdd == true. Sample:

fun foo(context: Context) {
    val bar = linearLayout(context) {
        view(
            width = 256,
            height = 128,
            onClick = { ... }
        )
    }
}
