Builds an instance of TextView by context and parameters or default. Sample:

fun foo(context: Context): TextView {
    return textView(
        context = context,
        layoutParams = ViewGroup::class.layoutParams(width = 512, height = 64),
        text = "bar",
        textSizeUnit = TypeDimension.PIXEL,
        textSize = 64 / 2f
    )
}
